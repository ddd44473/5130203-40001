

#include <iostream>
#include <cmath>
#include <cstdlib>
using namespace std;

int main() {
    int a, b, a1, b1;
    


    do {
        cout << "\n A = ";
        cin >> a;
        cout << "\n B = ";
        cin >> b;
    } while (a < 100 || a > 1000 || b < 100 || b > 1000);

    a1 = a;
    b1 = b;
    // calculate answer MCM - наименьшее общее кратное

    int i, mcm;
    i = 2;
    mcm = 1;

    while (a != 1 || b != 1) {
        if (a % i == 0 || b % i == 0) {
            mcm = mcm * i;

            if (a % i == 0)
                a = a / i;

            if (b % i == 0)
                b = b / i;
        }
        else {
            i++;
        }
    }
    // calculate MCD
    int j, mcd = 1;
    j = 2;
    while (a1 > j || b1 > j) {
        if (a1 % j == 0 && b1 % j == 0){
            mcd = mcd * j;
            a1 = a1 / j;
            b1 = b1 / j;
        }
        else {
            j++;
        }
        
    }

    cout << "\n MCM = " << mcm << endl;
    cout << "\n MCD = " << mcd << endl;


    return 0;
}
